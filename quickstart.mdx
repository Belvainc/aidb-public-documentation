## Setup your account and retrieve your API key

[Sign in](https://aidb.belva.ai) or [Create](https://aidb.belva.ai/signup) an account and navigate to the API Key page.

<Warning>
  Remember to keep you API keys in a secure place as it grants direct access to your data
</Warning>

### All that's left is to implement your desired endpoints

```typescript
import axios, { AxiosResponse } from "axios";

type ChatMsg = {
  role: string;
  content: string;
};

export const search = async ( value: string, page: number, limit: number, apiKey: string, chatHistory: Array<ChatMsg> = [] ): Promise<AxiosResponse> => {
  try {
    return await axios.post(
      process.env.NEXT_PUBLIC_API_URL + "/api/document/v1/search",
      {
        params: { search: value, limit: limit, page: page, chatHistory: chatHistory }
      },
      {
        headers: { Authorization: `API-Key ${apiKey}` },
      }
    );
  } catch (error) {
    return Promise.reject(error);
  }
};
```
